Interactive Map Demo

In the body of the page, add:

        <svg id="usa-map" />

        <script>
            InteractiveMapUSA.init({
                mapSVGPath: "/assets/usa-map.svg",
                maps: [
                    {
                        id: "usa-map",
                        linkTo: "https://en.wikipedia.org/wiki/{{state}}", // 'state' is the name of the state snake cased as in 'north_carolina'
                        links: {
                            NC: "https://www.youtube.com/watch?v=lqWxGE7zSPA" // state links can be overridden on a state by state basis; 'linkTo' will still be used if this configuration is not defined
                        },
                        enabledStates: ["NC", "NY"], // disabling states is also possible, but not together with 'enabledStates' configuration
                        styles: {
                            stateFill: "red",
                            disabledStateFill: "#782abb",
                            stateHoverFill: "#888fff"
                        }
                    }
                ]
            })
        </script>
